# --- CONFIGURATION MANAGER ---
class ConfigManager:
    def __init__(self, filename="config.json"):
        self.filename = filename
        self.defaults = {
            "chrome_path": "C:\\Program Files\\Google\\Chrome\\Application\\chrome.exe",
            "main_profile_name": "Default",
            "port": 9999,
            "data_timeout": 25,
            "phase_timeout": 300,
            "target_item_id": "64d9ecad463b7ffa9697e2cc",
            "target_item_type": "capsule"
        }
        self.config = self.defaults.copy()
        self.load()

    def load(self):
        if not os.path.exists(self.filename):
            print(f"[INFO] No config file found. Creating default '{self.filename}'.")
            self.save()
        else:
            try:
                with open(self.filename, 'r') as f: loaded_config = json.load(f)
                for key, value in self.defaults.items():
                    if key not in loaded_config: loaded_config[key] = value
                self.config = loaded_config
            except (json.JSONDecodeError, TypeError):
                print(f"[ERROR] Error reading '{self.filename}'. Recreating with defaults.")
                self.config = self.defaults.copy(); self.save()
        
        if not os.path.exists(self.get('chrome_path')):
            alt_path = "C:\\Program Files (x86)\\Google\\Chrome\\Application\\chrome.exe"
            if os.path.exists(alt_path):
                self.set('chrome_path', alt_path)
                self.save()

    def save(self):
        try:
            with open(self.filename, 'w') as f: json.dump(self.config, f, indent=4)
        except Exception as e:
            print(f"[ERROR] Could not save config file: {e}")

    def get(self, key):
        return self.config.get(key, self.defaults.get(key))

    def set(self, key, value):
        self.config[key] = value

# --- LOCAL CASE MANAGER ---
class CaseManager:
    def __init__(self, filename="cases.json"):
        self.filename = filename
        self.cases = []
        self.load_cases()

    def load_cases(self):
        if not os.path.exists(self.filename):
            log_message(f"'{self.filename}' not found. Case selection will be disabled.", "ERROR")
            self.cases = []
            return
        try:
            with open(self.filename, 'r', encoding='utf-8') as f:
                self.cases = json.load(f)
            self.cases.sort(key=lambda x: x.get('price', 0), reverse=True)
            log_message(f"Loaded {len(self.cases)} items from '{self.filename}'.", "SUCCESS")
        except (json.JSONDecodeError, Exception) as e:
            log_message(f"Error reading '{self.filename}': {e}", "ERROR")
            self.cases = []
            
    def get_cases(self):
        return self.cases

# --- SHARED UTILITIES ---
signal_queue = queue.Queue()
log_queue = queue.Queue()

class bcolors:
    OKGREEN = '\033[92m'
    WARNING = '\033[93m'
    FAIL = '\033[91m'
    ENDC = '\033[0m'
    HEADER = '\033[95m'

def log_message(m, l="INFO"):
    log_entry = f"[{datetime.now().strftime('%H:%M:%S')}] {m}"
    log_queue.put(log_entry)
    color = {"SUCCESS": bcolors.OKGREEN, "WARNING": bcolors.WARNING, "ERROR": bcolors.FAIL, "HEADER": bcolors.HEADER}.get(l, bcolors.ENDC)
    print(f"{color}{log_entry}{bcolors.ENDC}")

def format_number(n):
    try:
        return f"{int(n):,}"
    except (ValueError, TypeError):
        return str(n)

def ensure_profile_closed(p):
    command = f'wmic process where "name=\'chrome.exe\' and commandline like \'%--profile-directory={p}%\'" delete'
    subprocess.run(command, shell=True, stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL, creationflags=subprocess.CREATE_NO_WINDOW)

# --- WORKER THREADS ---
def build_url(config,phase,override_target=None):
    base="https://case-clicker.com/"; params={'runPhase':phase}
    target=override_target or {'id':config.get('target_item_id'),'type':config.get('target_item_type')}
    params['targetId']=target['id']; params['targetType']=target['type']
    page="trading" if phase=='3' else ""; return f"{base}{page}?{urlencode(params)}"
def process_single_phase(gui,config,p_name,phase,override_target=None):
    ensure_profile_closed(p_name); time.sleep(1)
    while not signal_queue.empty(): signal_queue.get_nowait()
    is_info=(phase=="getInfo"); timeout=config.get('data_timeout') if is_info else config.get('phase_timeout'); gui.add_profile_entry(p_name,{},f"Running Phase {phase}...");
    url=build_url(config,phase,override_target)
    proc=subprocess.Popen([config.get('chrome_path'),f"--profile-directory={p_name}",url],stdout=subprocess.DEVNULL,stderr=subprocess.DEVNULL)
    log_message(f"Launched '{p_name}' for phase '{phase}'. Waiting..."); success=False; timed_out=False
    try:
        while True:
            s = signal_queue.get(timeout=timeout)
            if s['type'] == 'heartbeat':
                gui.add_profile_entry(p_name, {}, f"Running Phase {phase} (Working...)")
                continue
            
            if s['type']=='info':
                p=s['payload'];
                if p not in ("LOGGED_OUT","ERROR"):
                    try: gui.add_profile_entry(p_name,json.loads(p),"Success"); success=True
                    except: gui.add_profile_entry(p_name,{},"Data Decode Error")
                else: gui.add_profile_entry(p_name,{},"Logged Out" if p=="LOGGED_OUT" else "Userscript Error")
            elif s['type']=='phase_success': gui.add_profile_entry(p_name,{},f"Phase {phase} Complete"); success=True
            elif s['type']=='phase_error': gui.add_profile_entry(p_name,{},f"Phase {phase} Failed")
            
            break

    except queue.Empty: log_message(f"Timeout for '{p_name}' phase '{phase}'.","ERROR"); gui.add_profile_entry(p_name,{},"Timed Out"); timed_out=True
    finally:
        if timed_out and proc.poll() is None: proc.terminate()
    return success
def process_trade_phase(gui,config,sender):
    ensure_profile_closed(sender); time.sleep(1); gui.add_profile_entry(sender,{},"Running Phase 3 (Send)...");
    while not signal_queue.empty(): signal_queue.get_nowait()
    proc_s=subprocess.Popen([config.get('chrome_path'),f"--profile-directory={sender}",build_url(config,"3")],stdout=subprocess.DEVNULL,stderr=subprocess.DEVNULL)
    log_message(f"Launched SENDER '{sender}'. Waiting for trade ID..."); t_id=None; success=False
    try:
        s=signal_queue.get(timeout=config.get('phase_timeout'));
        if s['type']=='phase_success':
            m=re.search(r"'tradeId': \['([0-9a-f]{24})'\]",s['payload'])
            if m: t_id=m.group(1); log_message(f"Trade by '{sender}'. ID: {t_id}","SUCCESS"); gui.add_profile_entry(sender,{},"Phase 3 (Sent)")
        else: log_message(f"Phase 3 failed for '{sender}'.","ERROR"); gui.add_profile_entry(sender,{},"Phase 3 (Send Failed)")
    except queue.Empty: log_message(f"Timeout for sender '{sender}'.","ERROR"); gui.add_profile_entry(sender,{},"Phase 3 (Send Timeout)")
    if proc_s.poll() is None and t_id is None: proc_s.terminate()
    if t_id:
        receiver=config.get('main_profile_name'); ensure_profile_closed(receiver); time.sleep(1); gui.add_profile_entry(receiver,{},"Running Phase 3 (Accept)...")
        proc_r=subprocess.Popen([config.get('chrome_path'),f"--profile-directory={receiver}",f"https://case-clicker.com/trading/{t_id}?autoJoin=1"],stdout=subprocess.DEVNULL,stderr=subprocess.DEVNULL)
        log_message(f"Launched RECEIVER '{receiver}' to accept trade. Waiting...");
        try:
            a_s=signal_queue.get(timeout=config.get('phase_timeout'))
            if a_s['type']=='phase_success': log_message(f"Trade {t_id} accepted.","SUCCESS"); gui.add_profile_entry(sender,{},"Phase 3 Complete"); success=True
            else: log_message(f"Error while '{receiver}' accepting.","ERROR"); gui.add_profile_entry(sender,{},"Phase 3 (Accept Failed)")
        except queue.Empty: log_message(f"Timeout for '{receiver}' to accept.","ERROR"); gui.add_profile_entry(sender,{},"Phase 3 (Accept Timeout)")
        if proc_r.poll() is None: proc_r.terminate()
    return success
def get_interactive_case_choice(gui):
    cases = gui.case_manager.get_cases()
    if not cases: log_message("Case list is empty or not loaded. Check cases.json.","ERROR"); return None,None
    popup=CaseSelectorWindow(gui.root,cases); gui.root.wait_window(popup)
    if popup.result: return popup.result[0],popup.result[1]
    return None,None
